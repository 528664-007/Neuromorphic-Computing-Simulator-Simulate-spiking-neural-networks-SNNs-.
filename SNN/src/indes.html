How It Works

The simulation models a Leaky Integrate-and-Fire (LIF) neuron. Adjust the weight and beta parameters to observe spiking behavior.

Python Code


import js
import json

def leaky_integrate_and_fire(mem, x, w, beta, threshold=1):
    spk = 1 if mem > threshold else 0
    mem = beta * mem + w * x - spk * threshold
    return spk, mem

def simulate_snn(weight=0.4, beta=0.819):
    num_steps = 200
    x = [0] * 10 + [0.5] * 190
    mem = 0.0
    mem_rec = []
    spk_rec = []

    for step in range(num_steps):
        spk, mem = leaky_integrate_and_fire(mem, x[step], w=weight, beta=beta)
        mem_rec.append(mem)
        spk_rec.append(spk)

    js.drawCanvas(json.dumps({"mem": mem_rec, "spk": spk_rec, "steps": num_steps}))
